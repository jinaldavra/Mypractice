#include<conio.h>
int main(){
	int num = 5,row,col;
	clrscr();

	for(row=1;row<=num;row++)
	{
		for(col=num;col>row;col--)
		{
			printf(" ");
		}
		for(col=num;col>=num-row+1;col--)
		{
			printf("* ");
		}
		printf("\n");
	}

}

#include<conio.h>
int main(){
	int num = 5,row,col;
	clrscr();

	for(row=1;row<=num;row++)
	{
		for(col=1;col<=num;col++)
		{
			if(row==col)
			{
				printf("* ");
			}else
			{
				printf("- ");
			}
		}
		printf("\n");
	}
	getch();
}
